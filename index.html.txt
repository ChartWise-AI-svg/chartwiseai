// Настрой Telegram бота
const telegramBotToken = 'YOUR_BOT_TOKEN'; // вставь сюда токен бота
const telegramChatId = 'YOUR_CHAT_ID'; // вставь сюда свой chat_id

document.getElementById('applyForm').addEventListener('submit', function(e){
e.preventDefault();

const name = document.getElementById('name').value;
const email = document.getElementById('email').value;
const portfolio = document.getElementById('portfolio').value;
const telegramUser = document.getElementById('telegram').value;
const message = document.getElementById('message').value;

// Формируем текст для Telegram
let text = `New Application:\nName: ${name}\nEmail: ${email}\nPortfolio: ${portfolio}\nMessage: ${message}`;
if(telegramUser) text += `\nTelegram: https://t.me/${telegramUser.replace('@','')}`;

// Отправка в Telegram
fetch(`https://api.telegram.org/bot${telegramBotToken}/sendMessage`, {
method: 'POST',
headers: {'Content-Type': 'application/json'},
body: JSON.stringify({ chat_id: telegramChatId, text: text })
})
.then(res => {
if(res.ok) alert('Application sent! ✅');
else alert('Error sending application.');
document.getElementById('applyForm').reset();
})
.catch(()=>alert('Error sending application.'));
});
</script>
